# Checkout Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a checkout flow where customers submit orders with Korean address lookup, orders are logged to Google Sheets, and confirmation emails are sent via Google Apps Script.

**Architecture:** New `page-checkout` SPA section with a two-column form/summary layout. Daum Postcode API for address lookup (script tag, no key). Google Apps Script web app handles both email and sheet logging via a single POST endpoint. No build tools — all vanilla JS/CSS/HTML.

**Tech Stack:** Vanilla JS, Daum Postcode API, Google Apps Script, GmailApp, Google Sheets

---

```javascript
function renderCheckoutSummary() {
  let subtotal = 0;
  let html = '';

  cart.forEach(item => {
    subtotal += item.price;
    let details = item.version;
    if (item.scanTube) details += ' / ' + t('opt_scantube') + ' ' + item.tubeLength;
    if (item.roller) details += ' / ' + t('opt_roller');
    if (item.film110) details += ' / 110';
    if (item.filmAPS) details += ' / APS';

    html += `
      <div class="checkout-item">
        <div>
          <div class="checkout-item-name">${item.name}</div>
          <div class="checkout-item-details">${details}</div>
        </div>
        <div class="checkout-item-price">₩${item.price.toLocaleString()}</div>
      </div>
    `;
  });

  checkoutItems.innerHTML = html;
  checkoutSubtotal.textContent = '₩' + subtotal.toLocaleString();
  checkoutTotal.textContent = '₩' + (subtotal + SHIPPING_FEE).toLocaleString();
}
```

```javascript
// ---- Form Validation ----
function validateCheckoutForm() {
  let valid = true;

  // Clear previous invalid states
  checkoutForm.querySelectorAll('input').forEach(input => input.classList.remove('invalid'));

  // Required fields
  const requiredFields = [custName, custPhone, custEmail, custZip, custAddress, custDetailAddress];
  for (const field of requiredFields) {
    if (!field.value.trim()) {
      field.classList.add('invalid');
      valid = false;
    }
  }

  // Phone format
  if (custPhone.value && !/^01[016789]-?\d{3,4}-?\d{4}$/.test(custPhone.value.trim())) {
    custPhone.classList.add('invalid');
    if (valid) showToast(t('checkout_invalid_phone'));
    valid = false;
  }

  // Email format
  if (custEmail.value && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(custEmail.value.trim())) {
    custEmail.classList.add('invalid');
    if (valid) showToast(t('checkout_invalid_email'));
    valid = false;
  }

  if (valid === false && !custPhone.classList.contains('invalid') && !custEmail.classList.contains('invalid')) {
    showToast(t('checkout_required'));
  }

  return valid;
}
```

- [ ] **Step 6: Add order submission**

```javascript
// ---- Order Submission ----
function generateOrderId() {
  const now = new Date();
  const pad = (n) => String(n).padStart(2, '0');
  return 'AD-' + now.getFullYear() + pad(now.getMonth() + 1) + pad(now.getDate()) + '-' + pad(now.getHours()) + pad(now.getMinutes()) + pad(now.getSeconds());
}

submitOrderBtn.addEventListener('click', async () => {
  if (!validateCheckoutForm()) return;

  submitOrderBtn.disabled = true;
  submitOrderBtn.textContent = t('checkout_submitting');

  const subtotal = cart.reduce((sum, item) => sum + item.price, 0);
  const total = subtotal + SHIPPING_FEE;
  const orderId = generateOrderId();

  const orderData = {
    orderId: orderId,
    name: custName.value.trim(),
    phone: custPhone.value.trim(),
    email: custEmail.value.trim(),
    zip: custZip.value.trim(),
    address: custAddress.value.trim(),
    detailAddress: custDetailAddress.value.trim(),
    items: cart.map(item => ({
      name: item.name,
      productKey: item.productKey,
      version: item.version,
      scanTube: item.scanTube,
      tubeLength: item.tubeLength,
      adapterSize: item.adapterSize,
      roller: item.roller,
      film110: item.film110,
      filmAPS: item.filmAPS,
      price: item.price,
    })),
    subtotal: subtotal,
    shipping: SHIPPING_FEE,
    total: total,
    lang: currentLang,
  };

  try {
    const response = await fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      mode: 'no-cors',
      headers: { 'Content-Type': 'text/plain' },
      body: JSON.stringify(orderData),
    });

    // no-cors means we can't read the response, so we assume success if no network error
    showConfirmation(orderId, total);
    cart = [];
    renderCart();
  } catch (err) {
    showToast(t('checkout_error'));
  } finally {
    submitOrderBtn.disabled = false;
    submitOrderBtn.textContent = t('checkout_submit');
  }
```

```javascript
const SPREADSHEET_ID = 'PASTE_YOUR_SPREADSHEET_ID_HERE';
const SHEET_NAME = 'Orders';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);

    // Write to spreadsheet
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow([
        'Order ID', 'Date', 'Name', 'Phone', 'Email',
        'Postal Code', 'Address', 'Detail Address',
        'Items', 'Subtotal', 'Shipping', 'Total', 'Payment Status'
      ]);
      sheet.getRange(1, 1, 1, 13).setFontWeight('bold');
    }

    const now = new Date();
    const dateStr = Utilities.formatDate(now, 'Asia/Seoul', 'yyyy-MM-dd HH:mm');

    sheet.appendRow([
      data.orderId,
      dateStr,
      data.name,
      data.phone,
      data.email,
      data.zip,
      data.address,
      data.detailAddress,
      JSON.stringify(data.items),
      data.subtotal,
      data.shipping,
      data.total,
      '미확인'
    ]);

    // Send confirmation email
    sendConfirmationEmail(data);

    return ContentService
      .createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
```
